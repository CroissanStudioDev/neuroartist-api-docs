```ts
// hono.ts
import { Hono } from 'hono'
import crypto from 'node:crypto'

const SECRET = process.env.NA_WEBHOOK_SECRET!.replace(/^wsec_/, '')
const app = new Hono()

app.post('/cb', async (c) => {
  const id = c.req.header('webhook-id')
  const ts = c.req.header('webhook-timestamp')
  const sig = c.req.header('webhook-signature')
  if (!id || !ts || !sig) return c.json({}, 400)

  if (Math.abs(Date.now() / 1000 - Number(ts)) > 300) return c.json({}, 401)

  const raw = await c.req.text()
  const key = Buffer.from(SECRET, 'base64')
  const expected = 'v1,' + crypto.createHmac('sha256', key)
    .update(`${id}.${ts}.${raw}`).digest('base64')

  const valid = sig.split(' ').some(s => {
    try { return crypto.timingSafeEqual(Buffer.from(s), Buffer.from(expected)) }
    catch { return false }
  })
  if (!valid) return c.json({}, 401)

  // Дедуп по webhook-id (24h)
  const seen = await redis.set(`wh:${id}`, '1', { NX: true, EX: 86400 })
  if (seen === null) return c.json({ ok: true, deduped: true })

  const event = JSON.parse(raw)
  if (event.type === 'generation.completed') await onCompleted(event)
  if (event.type === 'generation.failed')    await onFailed(event)

  return c.json({ ok: true })
})
```

```js
import express from 'express'
import crypto from 'node:crypto'

const SECRET = process.env.NA_WEBHOOK_SECRET.replace(/^wsec_/, '')
const app = express()

// IMPORTANT: raw body — НЕ express.json
app.post('/cb', express.raw({ type: 'application/json' }), (req, res) => {
  const id = req.header('webhook-id')
  const ts = req.header('webhook-timestamp')
  const sig = req.header('webhook-signature')
  if (!id || !ts || !sig) return res.status(400).end()

  if (Math.abs(Date.now() / 1000 - Number(ts)) > 300) return res.status(401).end()

  const raw = req.body.toString('utf8')
  const key = Buffer.from(SECRET, 'base64')
  const expected = 'v1,' + crypto.createHmac('sha256', key)
    .update(`${id}.${ts}.${raw}`).digest('base64')

  const valid = sig.split(' ').some(s => {
    try { return crypto.timingSafeEqual(Buffer.from(s), Buffer.from(expected)) }
    catch { return false }
  })
  if (!valid) return res.status(401).end()

  const event = JSON.parse(raw)
  res.json({ ok: true })
})
```

```python
import os, hmac, base64, time, json
from fastapi import FastAPI, Request, HTTPException

app = FastAPI()
SECRET = os.environ["NA_WEBHOOK_SECRET"].removeprefix("wsec_")

@app.post("/cb")
async def cb(req: Request):
    wid = req.headers.get("webhook-id")
    wts = req.headers.get("webhook-timestamp")
    wsig = req.headers.get("webhook-signature")
    if not (wid and wts and wsig):
        raise HTTPException(400)
    if abs(time.time() - int(wts)) > 300:
        raise HTTPException(401)

    raw = (await req.body()).decode()
    key = base64.b64decode(SECRET)
    expected = "v1," + base64.b64encode(
        hmac.new(key, f"{wid}.{wts}.{raw}".encode(), "sha256").digest()
    ).decode()

    if not any(hmac.compare_digest(s, expected) for s in wsig.split(" ")):
        raise HTTPException(401)

    event = json.loads(raw)
    return {"ok": True}
```

## Чек-лист

- [ ] Используется raw body (НЕ JSON middleware) до валидации.
- [ ] `webhook-timestamp` проверяется на свежесть (≤ 5 мин).
- [ ] Подпись сравнивается через `timingSafeEqual` / `compare_digest` / `hash_equals`.
- [ ] Дедуп по `webhook-id` (24h TTL).
- [ ] Endpoint отвечает `200 OK` за < 30 сек, тяжёлая работа уходит в очередь.
- [ ] При ошибке возвращается 5xx — gateway сам повторит до 6 раз.

## Retry-расписание

| Попытка | Задержка |
| --- | --- |
| 2 | 1 минута |
| 3 | 5 минут |
| 4 | 30 минут |
| 5 | 2 часа |
| 6 | 5 часов |

После 6-й — `webhook_dead` в `/me/activity`. Можно перечитать через `GET /queue/.../requests/{id}`.
